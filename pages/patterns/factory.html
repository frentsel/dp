The Factory Design Pattern In JavaScript

const ShapeFactory = function() {
  this.createShape = function(shapeType) {
    return new function() {
      this.draw = function() {
        console.log(`This is a ${shapeType}`);
      }
    };
  }
}

const factory = new ShapeFactory();

const rectangle = factory.createShape('rectangle');
const square = factory.createShape('square');
const triangle = factory.createShape('triangle');

rectangle.draw();
square.draw();
triangle.draw();
